int Impair(int n) {
    return n % 2 != 0;
}

    if (Impair(nombre))

    else

void max(int a,int b,int c,int d)
{
    if (a>b && b>c && c>d){
        printf("max est :%d",a);
    }
    else if (b>a && a>c && c>d){
        printf("max est :%d",b);
    }
    
    else if (c>a && a>b && b>d){
        printf("max est :%d",c);
    }
    
    
    else{
        printf("max est :%d",d);
    }
}
